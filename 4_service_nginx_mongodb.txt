#проверяем статус nginx
systemctl status nginx

#видим ошибку
nginx.service: Failed at step EXEC spawning /usr/sbin/nginxt: No such file or directory

#правим /lib/systemd/system/nginx.service , убираем линюю букву t -> nginxt
ExecStart=/usr/sbin/nginx -g 'daemon on; master_process on;'

systemctl daemon-reload
systemctl start nginx
